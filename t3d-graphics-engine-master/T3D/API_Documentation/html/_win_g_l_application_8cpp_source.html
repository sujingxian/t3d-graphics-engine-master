178  void WinGLApplication::handleEvent(SDL_Event *e){
179  switch (e->type) {
180  case SDL_QUIT:
181  running = false;
182  break;
183  case SDL_KEYDOWN:
184  Input::onKeyDown(e->key.keysym.sym);
185  if (Input::keysDown[KEY_ESCAPE])
186  running = false;
187  if (Input::keysDown[KEY_F1])
188  renderer->toggleWireframe();
189  if (Input::keysDown[KEY_F2])
190  renderer->toggleAxes();
191  if (Input::keysDown[KEY_F3])
192  renderer->toggleGrid();
193  if (Input::keysDown[KEY_F4])
194  renderer->togglePoints();
195  if (Input::keysDown[KEY_F9])
196  {
197  int line = 0;
198  addTask(new DiagMessageTask(this, "ESCAPE quit", 2, 600-(line++*20), true, 5.0));
199  addTask(new DiagMessageTask(this, "F1 wireframe", 2, 600-(line++*20), true, 5.0));
200  addTask(new DiagMessageTask(this, "F2 axes", 2, 600-(line++*20), true, 5.0));
201  addTask(new DiagMessageTask(this, "F3 grid", 2, 600 - (line++ * 20), true, 5.0));
202  addTask(new DiagMessageTask(this, "F4 points", 2, 600 - (line++ * 20), true, 5.0));
203  addTask(new DiagMessageTask(this, "F9 show help", 2, 600-(line++*20), true, 5.0));
204  addTask(new DiagMessageTask(this, "F10 show stats", 2, 600-(line++*20), true, 5.0));
205  }
206  if (Input::keysDown[KEY_F10])
207  {
208  // find log task
209  PerfLogTask *task = (PerfLogTask *)findTask("PerfLogTask");
210  if (task)
211  {
212  // toggle on screen diagnostics
213  task->setDiagDisplay(!task->getDiagDisplay());
214  }
215  }
216  break;
217  case SDL_KEYUP:
218  Input::onKeyUp(e->key.keysym.sym);
219  break;
220  case SDL_MOUSEMOTION:
221  Input::onMouseMotion((e->motion).xrel,(e->motion).yrel);
222  break;
223  }
224  }
225 
226 }
