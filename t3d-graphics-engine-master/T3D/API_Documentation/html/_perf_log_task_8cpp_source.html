67  void PerfLogTask::update(float dt){
68  frameCount++;
69  elapsedTime += dt;
70  sampleFrames++;
71  sampleElapsed += dt;
72 
73  if (sampleElapsed > PERF_SAMPLING_PERIOD) // update every quarter second
74  {
75  double currentFrameRate = sampleFrames/sampleElapsed;
76  double averageFrameRate = frameCount/elapsedTime;
77 
78  if (elapsedTime > 3.0 * PERF_SAMPLING_PERIOD) // allow some settling time
79  {
80  // Alternate way of calculating average frame rate
81  // This is mainly used to allow a settling time by avoiding frameCount and elapsedTime
82  sampleCount++;
83  frameRateTotal += currentFrameRate;
84  avgFrameRate = frameRateTotal / sampleCount;
85 
86  if (currentFrameRate > maxFrameRate)
87  maxFrameRate = currentFrameRate;
88  if (currentFrameRate < minFrameRate)
89  minFrameRate = currentFrameRate;
90  }
91 
92  if (diagDisplay)
93  {
94  font *f = app->getFont("resources/FreeSans.ttf", 16);
95  if (f != NULL)
96  {
97  std::ostringstream ss;
98  ss.precision(1);
99  ss << std::fixed;
100  ss << "frames: " << frameCount;
101  ss << ", elapsed time: " << elapsedTime;
102  ss << ", frame rate: min=" << minFrameRate << ", avg=" << averageFrameRate << ", max=" << maxFrameRate << ", cur=" << currentFrameRate << " (avg=" << avgFrameRate << ")";
103 
104  int w = 1024; // texture width, should be large enough for most diagnostics
105  int h = 32; // should be enough for single line (text wrap is not supported)
106  if (diagOverlay == NULL)
107  diagOverlay = new Texture(w,h);
108  diagOverlay->clear(Colour(0,0,0,255));
109  diagOverlay->writeText(0, 0, ss.str().c_str(), Colour(255,255,255,255), f->getFont());
110  app->getRenderer()->loadTexture(diagOverlay, false);
111 
112  if (!app->getRenderer()->exists2DOverlay(diagOverlay))
113  app->getRenderer()->add2DOverlay(diagOverlay, 0, 0);
114  }
115  }
116  else if (diagOverlay != NULL)
117  {
118  app->getRenderer()->remove2DOverlay(diagOverlay);
119  app->getRenderer()->unloadTexture(diagOverlay);
120  delete diagOverlay;
121  diagOverlay = NULL;
122  }
123 
124  sampleElapsed = 0;
