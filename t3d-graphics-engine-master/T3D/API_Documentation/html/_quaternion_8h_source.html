61  Quaternion(float theta_x, float theta_y, float theta_z)
62  {
63  float cos_z_2 = cosf(0.5f*theta_z);
64  float cos_y_2 = cosf(0.5f*theta_y);
65  float cos_x_2 = cosf(0.5f*theta_x);

67  float sin_z_2 = sinf(0.5f*theta_z);
68  float sin_y_2 = sinf(0.5f*theta_y);
69  float sin_x_2 = sinf(0.5f*theta_x);
70 
71  // and now compute Quaternion
72  s = cos_z_2*cos_y_2*cos_x_2 + sin_z_2*sin_y_2*sin_x_2;

75  v.z = sin_z_2*cos_y_2*cos_x_2 - cos_z_2*sin_y_2*sin_x_2;

101  Quaternion(const Matrix3x3 &m)
102  {
103  float t = m[0][0]+m[1][1]+m[2][2];
104  float r = sqrt(1+t);
105  s = 0.5f*r;

291  static Quaternion slerp(const Quaternion &q1, const Quaternion &q2, float t)
292  {
293  Quaternion q3;
294  float dot = Quaternion::dot(q1, q2);
295 
296  /* dot = cos(theta)
297  if (dot < 0), q1 and q2 are more than 90 degrees apart,
298  so we can invert one to reduce spinning */
299  if (dot < 0)
300  {
301  dot = -dot;
302  q3 = -q2;
303  } else q3 = q2;
304 
305  if (dot < 0.95f)
306  {
307  float angle = acosf(dot);
308  return (q1*sinf(angle*(1-t)) + q3*sinf(angle*t))/sinf(angle);
309  } else // if the angle is small, use linear interpolation
310  return lerp(q1,q3,t);
311  }

353  Vector3 eulerAngles() const
354  {
355  float sqw = s*s;
356  float sqx = v.x*v.x;
357  float sqy = v.y*v.y;
358  float sqz = v.z*v.z;
359 
360  float unit = sqx + sqy + sqz + sqw;
361  float test = v.x * v.y + v.z * s;
362 
363  Vector3 euler;
364 
365  if (test > 0.4999f * unit) // Singularity at north pole
366  {
367  euler.y = 2.f * (float)atan2f(v.x, s); // Yaw
368  euler.z = Math::PI * 0.5f; // Roll
369  euler.x = 0.f; // Pitch
370  return euler;
371  }
372  else if (test < -0.4999f * unit) // Singularity at south pole
373  {
374  euler.y = -2.f * (float)atan2f(v.x, s); // Yaw
375  euler.z = -Math::PI * 0.5f; // Roll
376  euler.x = 0.f; // Pitch
377  return euler;
378  }
379  else
380  {
381  euler.y = (float)atan2f(2.f * v.y * s - 2.f * v.x * v.z, sqx - sqy - sqz + sqw); // Yaw
382  euler.z = (float)asinf(2.f * test / unit); // Roll
383  euler.x = (float)atan2f(2.f * v.x * s - 2.f * v.y * v.z, -sqx + sqy - sqz + sqw); // Pitch
384  }
385  return euler;
386  }
