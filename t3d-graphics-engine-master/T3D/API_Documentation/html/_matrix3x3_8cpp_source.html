178  m[1][2]*m[2][1];
179  inv[0][1] = m[0][2]*m[2][1] -

181  inv[0][2] = m[0][1]*m[1][2] -
182  m[0][2]*m[1][1];
183  inv[1][0] = m[1][2]*m[2][0] -
184  m[1][0]*m[2][2];
185  inv[1][1] = m[0][0]*m[2][2] -
186  m[0][2]*m[2][0];
187  inv[1][2] = m[0][2]*m[1][0] -
188  m[0][0]*m[1][2];
189  inv[2][0] = m[1][0]*m[2][1] -
190  m[1][1]*m[2][0];
191  inv[2][1] = m[0][1]*m[2][0] -
192  m[0][0]*m[2][1];
193  inv[2][2] = m[0][0]*m[1][1] -
194  m[0][1]*m[1][0];
195 
196  float det =
197  m[0][0]*inv[0][0] +
198  m[0][1]*inv[1][0]+
199  m[0][2]*inv[2][0];

201  assert(abs(det) > tolerance);
202

204  for (int r = 0; r < 3; r++)
