224  void GLRenderer::loadMaterial(Material* mat){
225  if (mat != NULL){
226  glMaterialfv(GL_FRONT, GL_AMBIENT_AND_DIFFUSE, mat->getDiffuse());
227  glMaterialfv(GL_FRONT, GL_SPECULAR, mat->getSpecular());
228  glMaterialfv(GL_FRONT, GL_EMISSION, mat->getEmissive());
229  glMaterialf(GL_FRONT,GL_SHININESS, mat->getShininess());
230  glShadeModel(mat->getSmoothShading() ? GL_SMOOTH : GL_FLAT);
231  glDepthMask(mat->getDisablDepth() ? GL_FALSE : GL_TRUE); // enable/disable depth buffer write
232 
233  if (mat->getBlending() == Material::BlendMode::NONE) {
234  glDisable(GL_BLEND); // No Blending (although diffuse alpha will still be used)
235 
236  // enable "on/off" transparency ("cookie cutter alpha")
237  // This will only work for textures with an alpha channel (i.e. not bmp)
238  glAlphaFunc(GL_GREATER, 0.99f);
239  glEnable(GL_ALPHA_TEST);
240  }
241  else {
242  glDisable(GL_ALPHA_TEST);
243  glEnable(GL_BLEND); // Enable Blending
244 
245  // Material only supports a limited number of predefined blending modes
246  if (mat->getBlending() == Material::BlendMode::ADD) {
247  // colors are added (values >1 are clipped to 1)
248  glBlendFunc(GL_ONE, GL_ONE);
249  }
250  else if (mat->getBlending() == Material::BlendMode::MULTIPLY) {
251  // colors are multiplied (values >1 are clipped to 1)
252  glBlendFunc(GL_ZERO, GL_SRC_COLOR);
253  }
254  else {
255  // Assume Material::BlendMode::BLEND_DEFAULT
256  // transparency: alpha=0 - invisible, alpha=1 - no transparency
257  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
258  }
259  }
260 
261  if (mat->isTextured()){
262  glEnable(GL_TEXTURE_2D);
263  glBindTexture(GL_TEXTURE_2D,mat->getTexID());
264  glEnableClientState(GL_TEXTURE_COORD_ARRAY);
265 
266  glMatrixMode(GL_TEXTURE);
267  glLoadIdentity();
268  float s = mat->getTextureScale();
269  glScalef(s,s,s);
270  }
271  else {
272  glDisable(GL_TEXTURE_2D);
273  }
274 
275  Shader *shader = mat->getShader();
276  if (shader) {
277  shader->bindShader();
278  }
279  }
280  }

292  Mesh *mesh = object->getMesh();
293  if (mesh != NULL) {
294 
295  float *matdiffuse = NULL;
296  if (object->getAlpha() < 1.0)

300  if (object->getMaterial() != NULL) {

302  diffuse[0] = matdiffuse[0];

307  }
308 
309  glMatrixMode(GL_MODELVIEW);
310  glPushMatrix();
311  glMultTransposeMatrixf((object->getTransform()->getWorldMatrix()).getData());
312  drawMesh(mesh);
313  glPopMatrix();
314 
315  if (matdiffuse != NULL)
316  glMaterialfv(GL_FRONT, GL_AMBIENT_AND_DIFFUSE, matdiffuse); // restore material diffuse after alpha override
317 
318  }

397  // Helper function to get the GL texture format from a texture
398  int GLRenderer::getTextureFormat(Texture *tex){
399 
400  GLenum texture_format = GL_RGBA;
401  GLint nOfColors;
402 
403  // get the number of channels in the SDL surface
404  nOfColors = tex->getSurface()->format->BytesPerPixel;
405 
406  if (nOfColors == 4) // contains an alpha channel
407  {
408  if (tex->getSurface()->format->Rmask == 0x000000ff)
409  texture_format = GL_RGBA;
410  else
411  texture_format = GL_BGRA;
412  } else if (nOfColors == 3) // no alpha channel
413  {
414  if (tex->getSurface()->format->Rmask == 0x000000ff)
415  texture_format = GL_RGB;
416  else
417  texture_format = GL_BGR;
418  } else {
419  logger::Log(priority::Warning,
420  output_stream::All,
421  category::Video,
422  "Warning: loaded image is not truecolor.. this will probably break! Texture ID: %u\n"
423  ,
424  tex->getID());
425  }
426 
427  return texture_format;
428  }

431  void GLRenderer::loadTexture(Texture *tex, bool repeat){
432  GLuint textureID = 0; // This is a handle to our texture object
433  GLint minFilter, magFilter;
434 
435  // Check that the image's width is a power of 2
436  if ( (tex->getWidth() & (tex->getWidth() - 1)) != 0 ) {
437  logger::Log(priority::Warning,
438  output_stream::All,
439  category::Video,
440  "Warning: loaded texture's width is not a power of two. Texture ID: %u\n"
441  ,
442  tex->getID());
443  }
444 
445  // Also check if the height is a power of 2
446  if ( (tex->getHeight() & (tex->getHeight() - 1)) != 0 ) {
447  logger::Log(priority::Warning,
448  output_stream::All,
449  category::Video,
450  "Warning: loaded texture's height is not a power of two. Texture ID: %u\n"
451  ,
452  tex->getID());
453  }
454 
455  // Have OpenGL generate a texture object handle for us
456  glGenTextures( 1, &textureID );
457  tex->setID(textureID);
458 
459  // Bind the texture object
460  glBindTexture( GL_TEXTURE_2D, textureID );
461 
462  // Set the texture filtering
463  if (tex->getCountinuousTone()) {
464  // for a continuous-tone image we want bilinear interpolation
465  magFilter = GL_LINEAR;
466  if (tex->getMipmap()) {
467  // linear interpolation on the nearest mipmap for distant objects
468  minFilter = GL_LINEAR_MIPMAP_LINEAR;
469  }
470  else {
471  minFilter = GL_LINEAR;
472  }
473  }
474  else {
475  // a discrete-tone image, we want GL_NEAREST (so the texture doesn't appear out of focus)
476  minFilter = GL_NEAREST;
477  magFilter = GL_NEAREST;
478  }
479  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, minFilter );
480  glTexParameteri( GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, magFilter );
481 
482  // Set the texture's stretching properties
483  auto StretchingProperty = GL_REPEAT;
484  if (!repeat) StretchingProperty = GL_CLAMP_TO_EDGE;
485 
486  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, StretchingProperty);
487  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, StretchingProperty);
488 
489  // really loading for the first time but reload works if texture ID has been generated
490  reloadTexture(tex);
491  }

493  // Reload the texture from the SDL surface.
494  // It is assumed that the characteristics of the texture are unchanged
495  void GLRenderer::reloadTexture(Texture *tex)
496  {
497  GLuint textureID = 0; // This is a handle to our texture object
498  GLenum texture_format;
499  GLint nOfColors;
500 
501  textureID = tex->getID(); // texture must have been previously loaded with loadTexture
502 
503  // get the number of channels in the SDL surface
504  nOfColors = tex->getSurface()->format->BytesPerPixel;
505  // get the GL texture format
506  texture_format = getTextureFormat(tex);
507  const char *formatString;
508 
509  if (texture_format == GL_RGBA) {
510  formatString = "GL_RGBA";
511  } else if (texture_format == GL_RGB) {
512  formatString = "GL_RGB";
513 
514  } else if (texture_format == GL_BGRA) {
515  formatString = "GL_BGRA";
516  } else if (texture_format == GL_BGR) {
517  formatString = "GL_BGRA";
518  } else {
519  formatString = "Unknown?!";
520  }
521  logger::Log(priority::Tracing,
522  output_stream::File,
523  category::Video,
524  "Reloading texture.... Properties: "
525  "Mode: %s "
526  ,
527  formatString);
528 
529  // Bind the texture object
530  glBindTexture( GL_TEXTURE_2D, textureID );
531 
532  // Load the image
533  glTexImage2D(GL_TEXTURE_2D, 0, nOfColors, tex->getSurface()->w, tex->getSurface()->h, 0, texture_format, GL_UNSIGNED_BYTE, tex->getSurface()->pixels);
534 
535  if (tex->getMipmap()) {
536  // have GLU build mipmaps
537  int err = gluBuild2DMipmaps(GL_TEXTURE_2D, nOfColors, tex->getSurface()->w, tex->getSurface()->h, texture_format, GL_UNSIGNED_BYTE, tex->getSurface()->pixels);
538  if (err != 0) {
539  logger::Log(priority::Warning,
540  output_stream::All,
541  category::Video,
542  "Failed to generate mipmaps : %s", gluErrorString(err));
543  }
544  }
545  }

622  // based on code from http://www.gamedev.net/topic/284259-for-reference-using-sdl_ttf-with-opengl/
623  void GLRenderer::draw2DMesh(overlay2D *overlay)
624  {
625  int x = overlay->x;
626  int y = overlay->y;
627  int w = overlay->texture->getWidth();
628  int h = overlay->texture->getHeight();
629 
630  unsigned int texture = overlay->texture->getID();
631 
632  /* GL_NEAREST looks horrible, if scaled... */
633  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
634  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
635 
636  /* prepare to render our texture */
637  glEnable(GL_TEXTURE_2D);
638  glBindTexture(GL_TEXTURE_2D, texture);
639  glColor3f(1.0f, 1.0f, 1.0f);
640 
641  /* Draw a quad at location */
642  glBegin(GL_QUADS);
643  {
644  /* Recall that the origin is in the lower-left corner and in clip space / screen space, Y increases upwards.
645  That is why the TexCoords specify different corners than the Vertex coords seem to. */
646  float xLeft = float(x);
647  float xRight = float(x + w);
648  float yBottom = float(y);
649  float yTop = float(y + h);
650 
651  glTexCoord2f(0.0f, 1.0f); glVertex2f(xLeft, yBottom);
652  glTexCoord2f(1.0f, 1.0f); glVertex2f(xRight, yBottom);
653  glTexCoord2f(1.0f, 0.0f); glVertex2f(xRight, yTop);
654  glTexCoord2f(0.0f, 0.0f); glVertex2f(xLeft, yTop);
655  }
656  glEnd();
657 
658  glDisable(GL_TEXTURE_2D);
659 
660  glFinish();
661  }

664  // Show all diagnostic messages
665  // Messages are rendered and then deleted. A persistant message must be added every frame
666  // This isn't ideal but a better soloution would be to implement 2D sprites withing
667  void GLRenderer::showD2DOverlays()
668  {
669  if (overlays.empty()) return;
670 
671  // Orthographic projection view is fixed to the GL viewport size.
672  int vPort[4] = {0};
673  glGetIntegerv(GL_VIEWPORT, vPort);
674 
675  auto near = -1;
676  auto far = 1;
677  auto left = 0;
678  auto right = vPort[2];
679  auto bottom = 0;
680  auto top = vPort[3];
681 
682  // temp hardwired camera for overlay
683  Camera overlayCam = Camera(near, far, left, right, bottom, top);
684  setCamera(&overlayCam);
685 
686  glDisable(GL_DEPTH_TEST);
687  glDisable(GL_LIGHTING);
688  glDisable(GL_FOG);
689 
690  glEnable(GL_BLEND); // Enable Blending
691  //glBlendFunc(GL_SRC_COLOR,GL_ONE); // transparent black background (sort of)
692  glBlendFunc(GL_ONE,GL_ONE); // add overlay (works well for white text)
693 
694  for (auto &overlay : overlays)
695  {
696  glMatrixMode(GL_MODELVIEW);
697  glPushMatrix();
698  glLoadIdentity();
699 
700  draw2DMesh(overlay);
701  glPopMatrix();
702  }
703  glDisable(GL_BLEND);
704  }
