106  if (!numVerts)
107  {
108  logger::Log(priority::Warning,
109  output_stream::All,
110  category::Video,
111  "Called checkArrays() on an empty mesh!");
112 
113  return true;
114  }
115  uint32_t quadIndicesMax = numQuads ? (numQuads * 4u) - 1u : 0; /* prevent overflow */
116  uint32_t triIndicesMax = numTris ? (numTris * 3u) - 1u : 0; /* prevent overflow */
117 
118  bool ok = true;
119  for (uint32_t i = 0; i < numVerts; i++) {
120  Vector3 v = getVertex(i);
121  if (isnan(v.x) || isnan(v.y) || isnan(v.z)) {
122  logger::Log(priority::Warning,
123  output_stream::All,
124  category::Video,
125  "Vertex %u has not been set!", i);
126  ok = false;

270  // Calculate the normal vectors for each primitive that defines this mesh.
271  // *This sets each normal to be of unit length, i.e. normalized*.
272  // It also clears memory to zero before calculating normals.
273  // Note that for non-planar quads there may be lighting defects.
274  // This is especially noticable when using the Sweep class.
275  void Mesh::calcNormals(){
276  // set all normals to zero
277  for (uint32_t i = 0; i < numVerts; i++){
278  setNormal(i,0,0,0);
279  }
280 
281  // add normal for all tris
282  for (uint32_t i = 0; i < numTris; i++){
283  Vector3 v1 = getVertex(triIndices[i*3]);
284  Vector3 v2 = getVertex(triIndices[i*3+1]);
285  Vector3 v3 = getVertex(triIndices[i*3+2]);
286  Vector3 normal = (v2-v1).cross(v3-v2);
287  // normalise so that primitives of different sizes have equal weight
288  normal.normalise();
289  addNormal(triIndices[i*3],normal);
290  addNormal(triIndices[i*3+1],normal);
291  addNormal(triIndices[i*3+2],normal);
292  }
293 
294  // add normal for all quads
295  // nonplanar quads are given non-uniform normals to simulate curvature
296  // quads with 3 points collinear are treated as a tri
297  for (uint32_t i = 0; i < numQuads; i++) {
298  Vector3 vs[4];
299  for (int j = 0; j < 4; j++) {
300  vs[j] = getVertex(quadIndices[i * 4 + j]);
301  }
302  Vector3 normals[4];
303  bool isTri = false;
304  Vector3 triNormal = Vector3(0.0f, 0.0f, 0.0f);
305  for (int j = 0; j < 4; j++) {
306  Vector3 v1 = vs[(j+3)%4];
307  Vector3 v2 = vs[j];
308  Vector3 v3 = vs[(j+1)%4];
309  Vector3 normal = (v2 - v1).cross(v3 - v2);
310  if (normal.length() > 1e-08) {
311  // normalise so that primitives of different sizes have equal weight
312  normal.normalise();
313  normals[j] = normal;
314  triNormal = normal;
315  }
316  else {
317  isTri = true;
318  }
319  }
320  for (uint32_t j = 0; j < 4u; j++) {
321  addNormal(quadIndices[i*4+j], isTri ? triNormal : normals[j]);
322  }
323  }
324 
325  // normalise normals
326  normalise();
327  }

367  /*
368  * Given two offsets into a Vector3 representing the planes of interest,
369  * this function will find the min and max absolute values for each plane, then calculate and set
370  * the UVs for each vertex with respect to the min and max.
371  *
372  * \note Asserts 0 <= (vectorOffsetOne, vectorOffsetTwo) < 3.
373  *
374  */
375  void Mesh::calcUVPlane(uint32_t vectorOffsetOne, uint32_t vectorOffsetTwo)
376  {
377  assert(vectorOffsetOne < sizeof(Vector3) / sizeof(float));
378  assert(vectorOffsetTwo < sizeof(Vector3) / sizeof(float));
379 
380  float maxP1 = 0; // Maximum for 1st plane
381  float maxP2 = 0; // Maximum for 2nd plane
382 
383  for (uint32_t i = 0; i < numVerts; i++){
384  Vector3 v = getVertex(i);
385  float absP1 = fabs(v[vectorOffsetOne]);
386  float absP2 = fabs(v[vectorOffsetTwo]);
387 
388  maxP1 = std::max(absP1, maxP1);
389  maxP2 = std::max(absP2, maxP2);
390  }
391 
392  for (uint32_t i = 0; i < numVerts; i++){
393  Vector3 v = getVertex(i);
394 
395  float U = v[vectorOffsetOne] / maxP1 / 2.0f + 0.5f;
396  float V = v[vectorOffsetTwo] / maxP2 / 2.0f + 0.5f;
397 
398  setUV(i, U, V);
399  }
400  }
