89  Vector3 cameraPos;
90  float distance;
91  GameObject *object;
92 
93  // Single common camera for all rendering
94  if (camera)
95  cameraPos = camera->gameObject->getTransform()->getWorldPosition();
96  else
97  cameraPos = Vector3(0,0,0);
98 
99  buildRenderQueue(root);
100 
101  for (auto &priorities : materials) {
102  // temp list of objects requiring sorted draw order (if any)
103  std::priority_queue<GameObject*, std::vector<GameObject*>, GameObjectCameraDistanceCompare> sorted;
104 
105  for (auto &material : priorities){
106 
107  auto &q = material->getQueue(); // objects to be drawn
108 
109  if (!material->getSortedDraw()) {
110  // objects for normal unsorted materials are drawn immediately
111  loadMaterial(material);
112  while (!q.empty()) {
113  object = q.front();
114  if (object->isVisible()) {
115  draw(object);
116  }
117  q.pop();
118  }
119  unloadMaterial(material);
120  }
121  else {
122  // objects to be sorted and drawn later
123  while (!q.empty()) {
124  object = q.front();
125  if (object->isVisible()) {
126  // Note using squared distance as we only care about relative distance
127  distance = cameraPos.squaredDistance(object->getTransform()->getWorldPosition());
128  object->setDistanceToCamera(distance);
129  sorted.push(object);
130  }
131  q.pop();
132  }
133  }
134  }
135 
136  // Draw sorted objects (if any)
137  Material *loaded = NULL; // current loaded material
138  while (!sorted.empty()) {
139  object = sorted.top();
140  if (loaded != object->getMaterial()) {
141  // only load material if changed
142  loaded = object->getMaterial();
143  loadMaterial(loaded);
144  }
145  draw(object);
146  sorted.pop();
147  }
148  }
149  }
