49  /*
50  * \note T3D manages GameObject memory by destroying children of a scenegraph, *then* destroying the GameObject and its components.
51  */
52  Transform::~Transform(void)
53  {
54  // NOTE:
55  // This leaks memory, as removeChild() invalidates the size of the calling node's subtree.
56  // Given a std::vector of nodes, V, when `V.erase` is called on any index n, V will shuffle elements from [n + 1...V.size()].
57  // This shuffle is to ensure elements are densely contiguous in memory. It means that as the indices increase, half the scene graph
58  // is not deleted. It just happens to not segfalt because the condition `V.size()` is not inlined.
59  //
60  // A solution to this without a large refactoring of the component system would be to use std::shared_ptr or handles.
61  for (uint32_t child = 0; child < children.size(); child++) {
62  if (children[child]) delete children[child];
63  }
64 
65  if (parent) {
66  parent->removeChild(this); /* invalidates iterator of parent! */
67  parent = NULL;
68  }
69  children.clear();
70 
71  delete gameObject;
72  }

74  /*
75  * \param updateChildren Recursively update or not
76  */
77  void Transform::update(bool updateChildren)
78  {
79  if (needLocalUpdate) calcLocalMatrix();
80 
81  /* If this transform isn't the root, concatenate the parents matrix (operator '*' here)
82  with ours after checking if parent needs an update. Otherwise, this transform is
83  the root and is synonymous with world space. */
84  if (parent)
85  {
86  if (parent->needWorldUpdate) parent->update(false);
87  worldMatrix = parent->worldMatrix*localMatrix;
88  }
89  else
90  {
91  worldMatrix = localMatrix;
92  }
93 
94  needWorldUpdate = false;
95 
96  if(updateChildren)
97  {
98  for (auto child: children)
99  {
100  child->update(true);
101  }
102  }
103  }

264  /*
265  * This works by making -ve z axis point at other
266  *
267  * \param target transform to look at
268  * \param up upwards direction relative to transform
269  */
270  void Transform::lookAt(Vector3 target, Vector3 up){
271  Vector3 vectorTo = target - getWorldPosition();
272 
273  if (vectorTo.squaredLength()>0){
274  // first build rotation matrix
275  Vector3 zaxis = -vectorTo.normalised();
276  Vector3 xaxis = -zaxis.cross(up).normalised();
277  Vector3 yaxis = zaxis.cross(xaxis);
278  rotationMatrix.FromAxes(xaxis,yaxis,zaxis);
279  }
280  needLocalUpdate = true;
281  setNeedWorldUpdate();
282  }

409  void Transform::removeChild(Transform* c)
410  {
411  if(!c) return;
412 
413  for(unsigned int i = 0; i < children.size(); ++i)
414  {
415  if(children[i] == c)
416  {
417  children.erase(children.begin() + i);

419  }
420  }
