117  */
118  void DrawTask::flushPixelQueue()
119  {
120  const uint32_t MaxOutOfBoundsCount = 10u;
121  uint32_t OutOfBoundsCount = 0u;
122  for (auto &Pixel : pixelPlotQueue)
123  {
124  bool PixelWithinWidth = (Pixel.x >= 0 && Pixel.x < drawArea->getWidth());
125  bool PixelWithinHeight = (Pixel.y >= 0 && Pixel.y < drawArea->getHeight());
126  bool PixelWithinBounds = (PixelWithinWidth && PixelWithinHeight);
127 
128  if (PixelWithinBounds)
129  {
130  drawArea->plotPixel(Pixel.x, Pixel.y, Pixel.colour);
131  }
132  else
133  {
134  if (OutOfBoundsCount < MaxOutOfBoundsCount)
135  {
136  logger::Log(priority::Tracing,
137  output_stream::File,
138  category::Debug,
139  "Pixel out of bounds!\n"
140  "\tWidth :: [0 <= X <= %4u :: %4d :: %5s]%s\n"
141  "\tHeight :: [0 <= Y <= %4u :: %4d :: %5s]%s\n"
142  ,
143  app->getRenderer()->WindowWidth,
144  Pixel.x, PixelWithinWidth ? "OK" : "ERROR",
145  PixelWithinWidth ? "" : " <<<\n",
146  app->getRenderer()->WindowHeight,
147  Pixel.y, PixelWithinHeight ? "OK" : "ERROR",
148  PixelWithinHeight ? "" : " <<<\n");
149  }
