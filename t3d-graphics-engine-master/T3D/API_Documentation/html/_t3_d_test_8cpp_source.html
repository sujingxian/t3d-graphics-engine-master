37  bool T3DTest::init() {

41  //Create a skybox and add some fog
42  renderer->loadSkybox("Resources/Sunny1");
43  renderer->setFog(0.007f,0.8f,0.8f,0.8f,1.0f);
44  renderer->toggleFog();
45 
46  //Add a light
47  GameObject *lightObj = new GameObject(this);
48  Light *light = new Light(Light::Type::DIRECTIONAL);
49  light->setAmbient(1,1,1);
50  light->setDiffuse(1,1,1);
51  light->setSpecular(1,1,1);
52  lightObj->setLight(light);
53  lightObj->getTransform()->setLocalRotation(Vector3(-45*Math::DEG2RAD,70*Math::DEG2RAD,0));
54  lightObj->getTransform()->setParent(root);
55 
56  //Create some basic materials
57  Material *red = renderer->createMaterial(Renderer::PR_OPAQUE);
58  red->setDiffuse(1,0,0,1);
59  Material *green = renderer->createMaterial(Renderer::PR_OPAQUE);
60  green->setDiffuse(0,1,0,1);
61  Material *blue = renderer->createMaterial(Renderer::PR_OPAQUE);
62  blue->setDiffuse(0,0,1,1);
63 
64  //Create some textured materials (using texture files)
65  Texture *smileytex = new Texture("Resources/Smiley.bmp", true, true);
66  renderer->loadTexture(smileytex);
67  Material *smiley = renderer->createMaterial(Renderer::PR_OPAQUE);
68  smiley->setTexture(smileytex);
69 
70  Texture *sparkletex = new Texture("Resources/sparkle.png", true, true);
71  renderer->loadTexture(sparkletex);
72  Material *sparklemat = renderer->createMaterial(Renderer::PR_TRANSPARENT);
73  sparklemat->setEmissive(1, 1, 1, 1);
74  sparklemat->setBlending(Material::BlendMode::DEFAULT);
75  sparklemat->setSortedDraw(true, false);
76  sparklemat->setTexture(sparkletex);
77 
78  //Create a textured material by adding text
79  Texture *texttex = new Texture(128,32);
80  texttex->clear(Colour(255,255,255,255));
81  font *f = getFont("resources/FreeSans.ttf", 24);
82  if (f != NULL) {
83  texttex->writeText(2, 0, "Hello", Colour(0,255,0,255), f->getFont());
84  texttex->writeText(64, 0, "World", Colour(255,0,0,0), f->getFont());
85  }
86  renderer->loadTexture(texttex, true);
87  Material *textmat = renderer->createMaterial(Renderer::PR_OPAQUE);
88  textmat->setTexture(texttex,1);
89  textmat->setEmissive(1,1,1,1);
90 
91  //Create a material with a procedurally generated texture
92  Texture *proctex = new Texture(512,512);
93  proctex->createFractal(Colour(40,150,50,255),Colour(120,220,100,255),25.0f,false);
94  renderer->loadTexture(proctex, true);
95  Material *procmat = renderer->createMaterial(Renderer::PR_TERRAIN);
96  procmat->setTexture(proctex,50);
97  procmat->setSpecular(0,0,0,0);
98 
99  //Create a camera
100  GameObject *camObj = new GameObject(this);
101  renderer->camera = new Camera(0.3, 300.0, 45.0, 1.6);
102  camObj->getTransform()->setLocalPosition(Vector3(0,0,20));
103  camObj->getTransform()->setLocalRotation(Vector3(0,0,0));
104  camObj->setCamera(renderer->camera);
105  camObj->getTransform()->setParent(root);
106 
107  //Add a controller to the camera object
108  camObj->addComponent(new KeyboardController());
109 
110  //Add some terrain
111  GameObject *terrain = new GameObject(this);
112  Terrain *terrainComponent = new Terrain();
113  terrain->addComponent(terrainComponent);
114  terrainComponent->createFractalTerrain(256,500,15,2.0); // procedurally generate terrain
115  terrain->setMaterial(procmat);
116  terrain->getTransform()->setLocalPosition(Vector3(0,-20,0));
117  terrain->getTransform()->setParent(root);
118  terrain->getTransform()->name = "Terrain";
119 
120  //Add a welcome message
121  GameObject *billboard = new GameObject(this);
122  Billboard *billboardComponent = new Billboard(renderer->camera->gameObject->getTransform(),true);
123  billboard->addComponent(billboardComponent);
124  billboard->setMaterial(textmat); // hello world
125  billboard->getTransform()->setLocalPosition(Vector3(0,3,0));
126  billboard->getTransform()->setLocalScale(Vector3(6,1.5,1));
127  billboard->getTransform()->setParent(root);
128  billboard->getTransform()->name = "Billboard";
129 
130  //Add a cube mesh
131  GameObject *cube = new GameObject(this);
132  cube->setMesh(new Cube(1));
133  cube->setMaterial(smiley);
134  cube->getTransform()->setLocalPosition(Vector3(4,-3,0));
135  cube->getTransform()->setParent(root);
136  cube->getTransform()->name = "Cube";
137 
138  //Create an empty node to use as a rotation point
139  GameObject *rotateOrigin = new GameObject(this);
140  rotateOrigin->getTransform()->setParent(root);
141  rotateOrigin->addComponent(new RotateBehaviour(Vector3(0,1,0)));
142 
143  //Create a torus using the Sweep class as a child of rotateOrigin
144  SweepPath sp;
145  sp.makeCirclePath(2,32);
146  GameObject *torus = new GameObject(this);
147  std::vector<Vector3> points;
148  points.push_back(Vector3(0.2f,0.0f,0.0f));
149  points.push_back(Vector3(0.14f,0.14f,0.0f));
150  points.push_back(Vector3(0.0f,0.2f,0.0f));
151  points.push_back(Vector3(-0.14f,0.14f,0.0f));
152  points.push_back(Vector3(-0.2f,0.0f,0.0f));
153  points.push_back(Vector3(-0.14f,-0.14f,0.0f));
154  points.push_back(Vector3(0.0f,-0.2f,0.0f));
155  points.push_back(Vector3(0.14f,-0.14f,0.0f));
156  torus->setMesh(new Sweep(points,sp,true));
157  torus->setMaterial(red);
158  torus->getTransform()->setLocalPosition(Vector3(10,0,0));
159  torus->getTransform()->setParent(rotateOrigin->getTransform());
160  torus->getTransform()->name = "Torus";
161 
162  //Add a sphere mesh as a child of the torus
163  GameObject *sphere = new GameObject(this);
164  sphere->setMesh(new Sphere(0.5,32));
165  sphere->setMaterial(blue);
166  sphere->getTransform()->setLocalPosition(Vector3(0,5,0));
167  sphere->getTransform()->setParent(torus->getTransform());
168  sphere->getTransform()->name = "Sphere";
169 
170  //Create some animation for the sphere and torus
171  Animation *anim = new Animation(10.0);
172  torus->addComponent(anim);
173  anim->addKey("Sphere",0,Quaternion(),Vector3(0,5,0));
174  anim->addKey("Sphere",5.0,Quaternion(),Vector3(0,-5,0));
175  anim->addKey("Sphere",7.0,Quaternion(),Vector3(0,0,0));
176  anim->addKey("Sphere",10.0,Quaternion(),Vector3(0,5,0));
177  anim->addKey("Torus",0,Quaternion(),Vector3(10,0,0));
178  anim->addKey("Torus",5.0,Quaternion(Vector3(0,0,Math::HALF_PI)),Vector3(5,0,0));
179  anim->addKey("Torus",7.0,Quaternion(),Vector3(15,0,0));
180  anim->addKey("Torus",10.0,Quaternion(),Vector3(10,0,0));
181  anim->loop(true);
182  anim->play();
183 
184  //Add a cube to act as a source for particle system
185  GameObject *cubeF = new GameObject(this);
186  cubeF->setMesh(new Cube(1));
187  cubeF->setMaterial(red);
188  cubeF->getTransform()->setLocalPosition(Vector3(-3, -3, 0));
189  cubeF->getTransform()->setLocalScale(Vector3(0.5f, 0.5f, 0.5f));
190  cubeF->getTransform()->setParent(root);
191  cubeF->getTransform()->name = "Fountain";
192 
193  // A simple fireworks fountain using the particle system
194  ParticleEmitter *particleSys = new ParticleEmitter(10.0f, 1.0f, 20.0f, 2.0f, 3.0f, 3.0f);
195  cubeF->addComponent(particleSys); // make cube source of particles
196 
197  // Create a pool of particles
198  for (int i = 0; i<100; i++)
199  {
200  GameObject *particle = new GameObject(this);
201  Billboard *bbComponent = new Billboard(camObj->getTransform(), true);
202  particle->addComponent(bbComponent);
203 
204  // start gravity particle with random velocity (mostly upwards)
205  ParticleGravity *behaviour = new ParticleGravity(particleSys,
206  Vector3(Math::randRange(-2,2), Math::randRange(5,9), Math::randRange(-2,2)), -9.8f, 1.2f);
207  behaviour->setAlphaFade(1.0f, 0.1f); // fade out particle during lifespan
208  particle->addComponent(behaviour); // add to particle system
209 
210  particle->setMaterial(sparklemat);
211  particle->getTransform()->setLocalScale(Vector3(0.8f, 0.8f, 0.8f)); // make them a bit smaller
212  particle->getTransform()->setParent(root); // particles movement independent of emitter
213  particle->getTransform()->name = "particle";
214 
215  particleSys->addParticle(behaviour, false);
216  }
217  particleSys->emit(3); // emit some particles to start
218 
219 
220  //Add a behaviour to the Cube to make it "look at" the sphere
221  cube->addComponent(new LookAtBehaviour(sphere->getTransform()));
222 
223  //Add a help overlay
224  addTask(new DiagMessageTask(this, "Press F9 for help", 2, 32, true, 5.0));
225 
226  return true;
227  }
