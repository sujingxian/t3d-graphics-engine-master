34  void SoundManager::init() {
35  logger::Log(priority::Info,
36  output_stream::All,
37  category::Audio,
38  "Initialising SoundManager...");
39 
40  FMOD_RESULT result;
41  unsigned int version;
42  int numdrivers;
43 #pragma warning(suppress: 26812) /* unscoped enum warning */
44  FMOD_SPEAKERMODE speakermode;
45  FMOD_CAPS caps;
46  char name[256];
47  /*
48  Create a System object and initialize.
49  */
50  result = FMOD::System_Create(&system);
51  errorCheck(result);
52  result = system->getVersion(&version);
53  errorCheck(result);
54 
55  if (version < FMOD_VERSION)
56  {
57  logger::Log(priority::Error,
58  output_stream::File,
59  category::Audio,
60  "Error! You are using an old version of FMOD %08x. This program requires %08x.",
61  version, FMOD_VERSION);
62  }
63 
64  result = system->getNumDrivers(&numdrivers);
65  errorCheck(result);
66 
67  if (numdrivers == 0)
68  {
69  result = system->setOutput(FMOD_OUTPUTTYPE_NOSOUND);
70  errorCheck(result);
71  }
72  else
73  {
74  result = system->getDriverCaps(0, &caps, 0, &speakermode);
75  errorCheck(result);
76  /*
77  Set the user selected speaker mode.
78  */
79  result = system->setSpeakerMode(speakermode);
80  errorCheck(result);
81  if (caps & FMOD_CAPS_HARDWARE_EMULATED)
82  {
83  /*
84  The user has the 'Acceleration' slider set to off! This is really bad for latency!
85  */
86  logger::Log(priority::Warning,
87  output_stream::File,
88  category::Audio,
89  "FMOD is unable to use hardware acceleration! You may notice significant audio latency.");
90  result = system->setDSPBufferSize(1024, 10);
91  errorCheck(result);
92  }
93  result = system->getDriverInfo(0, name, 256, 0);
94  errorCheck(result);
95  if (strstr(name, "SigmaTel"))
96  {
97  /*
98  Sigmatel sound devices crackle for some reason if the format is PCM 16bit.
99  PCM floating point output seems to solve it.
100  */
101  logger::Log(priority::Info,
102  output_stream::File,
103  category::Audio,
104  "SigmaTel sound device detected. Changing PCM output to 16-bit float to (hopefully) prevent crackling.");
105 
106  result = system->setSoftwareFormat(48000, FMOD_SOUND_FORMAT_PCMFLOAT, 0,0,
107  FMOD_DSP_RESAMPLER_LINEAR);
108  errorCheck(result);
109  }
110  }
111  result = system->init(100, FMOD_INIT_NORMAL, 0);
112  if (result == FMOD_ERR_OUTPUT_CREATEBUFFER)
113  {
114  /*
115  Ok, the speaker mode selected isn't supported by this soundcard. Switch it
116  back to stereo...
117  */
118  logger::Log(priority::Info,
119  output_stream::File,
120  category::Audio,
121  "Speaker mode unsupported by soundcard. Trying stereo mode...");
122  result = system->setSpeakerMode(FMOD_SPEAKERMODE_STEREO);
123  errorCheck(result);
124 
125  /*
126  ... and re-init.
127  */
128  result = system->init(100, FMOD_INIT_NORMAL, 0);
129  }
130  errorCheck(result);
131  }
