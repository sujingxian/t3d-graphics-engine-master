69  Vector3 Plane::projectVector(const Vector3& p) const
70  {
71  // We know plane normal is unit length, so use simple method
72  Matrix3x3 xform;
73  xform[0][0] = 1.0f - normal.x * normal.x;
74  xform[0][1] = -normal.x * normal.y;
75  xform[0][2] = -normal.x * normal.z;
76  xform[1][0] = -normal.y * normal.x;
77  xform[1][1] = 1.0f - normal.y * normal.y;
78  xform[1][2] = -normal.y * normal.z;
79  xform[2][0] = -normal.z * normal.x;
80  xform[2][1] = -normal.z * normal.y;
81  xform[2][2] = 1.0f - normal.z * normal.z;
