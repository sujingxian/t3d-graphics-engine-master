88  void ParticleEmitter::update(float dt)
89  {
90  float count;
91  elapsed += dt; // total particle system run time
92 
93  if (elapsed < duration)
94  {
95  // work out how many particles should have been emitted for the current elapsed time
96  // Includes ramp up, run and wind down as appropriate
97 
98  // Get expected particles for initial ramp up
99  count = emitRamp(startRate, emitRate, rampUpDuration, elapsed, 0.1f);
100 
101  // Plus particles for constant run
102  count += emitRamp(emitRate, emitRate, duration - (rampUpDuration+rampDownDuration), elapsed - rampUpDuration, 0.1f);
103 
104  // Plus particles for wind down
105  count += emitRamp(emitRate, endRate, rampDownDuration, elapsed - (duration - rampDownDuration), 0.1f);
106 
107  //std::cout << "expected particles for time " << elapsed << " is " << count << ", particles emitted so far " << emitted << std::endl;
108 
109  // count is now the total number of particles (with some random variability) that should have
110  // been generated by this time in the particles systems life.
111  count -= emitted; // minus the number already emitted to give how many we need to emit now
112 
113  int e = (int)(count + 0.5f); // emit rounded up output count
114  emit(e); // emit particles
115  emitted += e; // total emitted so far
116  }
117  }

127  float ParticleEmitter::emitRamp(float start, float end, float duration, float time, float variability)
128  {
129  float accel; // calculated acceration rate for ramp
130  float count; // expected number of particles produced to the given time
131 
132  if (time < 0 || duration <= 0) return 0; // not in ramp or no ramp
133 
134  if (time > duration) time = duration; // clamp time to max duration of ramp
135 
136  accel = (end - start) / duration; // particles per second per second
137  count = start * time + (accel * time * time) / 2.0f;
138  count += count * (Math::randRange(0, variability*2) - variability);
139 
140  return count;
141  }
