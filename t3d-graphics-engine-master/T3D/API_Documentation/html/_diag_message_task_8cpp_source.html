62  /*
63  * \param dt Change in time
64  * \note If this function is not working, it's probably a font error, but it could also be a hard-coded screen coordinate. Check `T3D_Log.txt`.
65  */
66  void DiagMessageTask::update(float dt){
67  if (timed)
68  {
69  time -= dt;
70  if (time < 0.0f) setFinished(true);
71  }
72 
73  if (refresh)
74  {
75  removeOverlay();
76 
77  font *f = app->getFont("Resources/FreeSans.ttf", 16);
78  if (f)
79  {
80  if (!diagOverlay) {
81  // Create and add overlay
82  int w = app->getRenderer()->WindowWidth; // texture width, should be large enough for most diagnostics
83  int h = 32; // should be enough for single line (text wrap is not supported)
84 
85  diagOverlay = new Texture(w,h);
86 
87  diagOverlay->clear(Colour(0,0,0,255));
88  diagOverlay->writeText(0, 0, message.c_str(), Colour(255,255,255,255), f->getFont());
89  app->getRenderer()->loadTexture(diagOverlay, false); // load new texture in renderer
90  app->getRenderer()->add2DOverlay(diagOverlay, x, y); // add to renderer overlays
91 
92  logger::Log(priority::Tracing,
93  output_stream::File,
94  category::Task,
95  "Created overlay for DiagMessageTask");
96  }
97  else
98  {
99  // refresh overlay
100  diagOverlay->clear(Colour(0,0,0,255));
101  diagOverlay->writeText(0, 0, message.c_str(), Colour(255,255,255,255), f->getFont());
102  app->getRenderer()->reloadTexture(diagOverlay); // reload texture in renderer
103  }
104  }
105  }
